"PyAutoGUI와 Tkinter를 사용해서 다음 기능을 가진 스크린샷 자동화 도구를 만들어줘: 1. 사용자가 화면의 특정 영역을 선택할 수 있는 기능 2. 스크린샷 간격(초 단위)을 설정할 수 있는 입력 필드 3. 스크린샷 저장 폴더 선택 기능 4. 시작/정지 버튼 5. 상태 표시 영역"

import tkinter as tk
from tkinter import filedialog, messagebox
import pyautogui
import time
import os
from datetime import datetime
import threading

class ScreenshotApp:
    def __init__(self, root):
        self.root = root
        self.root.title("자동 스크린샷 도구")
        self.root.geometry("500x400")
        self.root.resizable(False, False)
        
        self.running = False
        self.thread = None
        
        # 영역 선택 변수
        self.region = {"x1": 0, "y1": 0, "x2": 0, "y2": 0}
        self.save_path = os.path.join(os.path.expanduser("~"), "Screenshots")
        
        self.setup_ui()
    
    def setup_ui(self):
        # 영역 선택 프레임
        region_frame = tk.LabelFrame(self.root, text="스크린샷 영역")
        region_frame.pack(fill="x", padx=10, pady=10)
        
        tk.Button(region_frame, text="영역 선택하기", 
                 command=self.select_region).pack(pady=10)
        
        self.region_label = tk.Label(region_frame, 
                                    text="선택된 영역: 없음")
        self.region_label.pack(pady=5)
        
        # 설정 프레임
        settings_frame = tk.LabelFrame(self.root, text="설정")
        settings_frame.pack(fill="x", padx=10, pady=10)
        
        tk.Label(settings_frame, text="스크린샷 간격(초):").grid(
            row=0, column=0, padx=5, pady=5, sticky="w")
        self.interval_entry = tk.Entry(settings_frame, width=10)
        self.interval_entry.insert(0, "5")
        self.interval_entry.grid(row=0, column=1, padx=5, pady=5)
        
        tk.Label(settings_frame, text="저장 경로:").grid(
            row=1, column=0, padx=5, pady=5, sticky="w")
        self.path_label = tk.Label(settings_frame, 
                                 text=self.save_path, 
                                 anchor="w")
        self.path_label.grid(row=1, column=1, padx=5, pady=5, sticky="w")
        tk.Button(settings_frame, text="찾아보기", 
                command=self.select_path).grid(
                    row=1, column=2, padx=5, pady=5)
        
        # 제어 프레임
        control_frame = tk.Frame(self.root)
        control_frame.pack(fill="x", padx=10, pady=20)
        
        self.start_button = tk.Button(control_frame, text="시작", 
                                     command=self.start_screenshots,
                                     bg="#4CAF50", fg="white", width=10)
        self.start_button.pack(side="left", padx=10)
        
        self.stop_button = tk.Button(control_frame, text="정지", 
                                   command=self.stop_screenshots,
                                   bg="#F44336", fg="white", width=10,
                                   state="disabled")
        self.stop_button.pack(side="left", padx=10)
        
        # 상태 표시 영역
        status_frame = tk.LabelFrame(self.root, text="상태")
        status_frame.pack(fill="both", expand=True, padx=10, pady=10)
        
        self.status_text = tk.Text(status_frame, height=8, width=50)
        self.status_text.pack(padx=5, pady=5, fill="both", expand=True)
        self.status_text.config(state="disabled")
        
    def update_status(self, message):
        self.status_text.config(state="normal")
        self.status_text.insert(tk.END, 
                              f"[{datetime.now().strftime('%H:%M:%S')}] {message}\n")
        self.status_text.see(tk.END)
        self.status_text.config(state="disabled")
    
    def select_region(self):
        self.root.iconify()  # 창 최소화
        time.sleep(0.5)
        self.update_status("영역을 선택하세요...")
        
        try:
            region = pyautogui.screenshot()
            x1, y1, x2, y2 = pyautogui.dragRel.box()
            self.region = {"x1": x1, "y1": y1, 
                          "x2": x2, "y2": y2,
                          "width": x2-x1, "height": y2-y1}
            self.region_label.config(
                text=f"선택된 영역: ({x1}, {y1}) - ({x2}, {y2})")
            self.update_status(f"영역 선택 완료: {x2-x1}x{y2-y1} 픽셀")
        except Exception as e:
            self.update_status(f"영역 선택 오류: {str(e)}")
        
        self.root.deiconify()  # 창 복원
    
    def select_path(self):
        path = filedialog.askdirectory()
        if path:
            self.save_path = path
            self.path_label.config(text=path)
            self.update_status(f"저장 경로 변경: {path}")
    
    def take_screenshots(self):
        interval = float(self.interval_entry.get())
        count = 1
        
        while self.running:
            try:
                if all(self.region.values()):
                    filename = f"screenshot_{datetime.now().strftime('%Y%m%d_%H%M%S')}.png"
                    full_path = os.path.join(self.save_path, filename)
                    
                    img = pyautogui.screenshot(region=(
                        self.region["x1"], self.region["y1"],
                        self.region["width"], self.region["height"]))
                    img.save(full_path)
                    
                    self.update_status(f"스크린샷 #{count} 저장: {filename}")
                    count += 1
                else:
                    self.update_status("오류: 영역이 선택되지 않았습니다.")
                    self.stop_screenshots()
                    break
            except Exception as e:
                self.update_status(f"오류 발생: {str(e)}")
            
            time.sleep(interval)
    
    def start_screenshots(self):
        if not os.path.exists(self.save_path):
            os.makedirs(self.save_path)
        
        try:
            float(self.interval_entry.get())
        except ValueError:
            messagebox.showerror("오류", "유효한 간격(초)을 입력하세요.")
            return
        
        self.running = True
        self.thread = threading.Thread(target=self.take_screenshots)
        self.thread.daemon = True
        self.thread.start()
        
        self.start_button.config(state="disabled")
        self.stop_button.config(state="normal")
        self.update_status("스크린샷 자동화 시작됨")
    
    def stop_screenshots(self):
        self.running = False
        self.start_button.config(state="normal")
        self.stop_button.config(state="disabled")
        self.update_status("스크린샷 자동화 중지됨")

if __name__ == "__main__":
    root = tk.Tk()
    app = ScreenshotApp(root)
    root.mainloop()
