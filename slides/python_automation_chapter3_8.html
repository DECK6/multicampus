데이터 구조와 제어문 통합 활용

조건문과 반복문을 데이터 구조와 함께 사용하여 더 강력한 업무 자동화 솔루션을 구축하는 방법을 알아봅니다.

엑셀 데이터 분석 자동화

여러 엑셀 파일에서 데이터를 추출하고 조건에 맞게 필터링하여 분석 보고서를 자동으로 생성하는 예제입니다.

여러 엑셀 파일을 읽고 조건에 맞는 데이터만 추출하는 코드를 작성해줘
import pandas as pd
import os
import glob
# 모든 엑셀 파일 경로 가져오기
파일_경로 = glob.glob("./판매데이터/*.xlsx")
# 결합할 데이터프레임 초기화
결합_데이터 = pd.DataFrame()
for 파일 in 파일_경로:
파일명 = os.path.basename(파일)
print(f"처리 중: {파일명}")
# 엑셀 파일 읽기
df = pd.read_excel(파일)
# 조건에 맞는 데이터만 필터링 (판매액 1만원 이상)
필터링_데이터 = df[df['판매액'] >= 10000]
# 파일 정보 추가
필터링_데이터['파일출처'] = 파일명
# 결합 데이터프레임에 추가
결합_데이터 = pd.concat([결합_데이터, 필터링_데이터])
# 결과 요약
print(f"총 {len(결합_데이터)}개 행이 추출되었습니다.")
# 부서별 판매액 합계
부서별_합계 = 결합_데이터.groupby('부서')['판매액'].sum()
print(부서별_합계)

조건부 데이터 처리 자동화

데이터의 조건에 따라 서로 다른 처리 로직을 적용하는 자동화 예제입니다.

고객 데이터를 세그먼트별로 분류하고 다른 처리를 하는 코드를 작성해줘
import pandas as pd
# 고객 데이터 로드
고객_df = pd.DataFrame({
'고객번호': [1001, 1002, 1003, 1004, 1005],
'이름': ['김고객', '이고객', '박고객', '최고객', '정고객'],
'구매액': [50000, 120000, 35000, 200000, 80000],
'방문횟수': [2, 8, 1, 5, 3]
})
# 고객 세그먼트 분류 함수
def 고객_분류(row):
if row['구매액'] >= 100000 and row['방문횟수'] >= 5:
return 'VIP'
elif row['구매액'] >= 50000 or row['방문횟수'] >= 3:
return '우수고객'
else:
return '일반고객'
# apply로 각 행에 함수 적용
고객_df['세그먼트'] = 고객_df.apply(고객_분류, axis=1)
# 세그먼트별 마케팅 메시지 생성
for index, row in 고객_df.iterrows():
if row['세그먼트'] == 'VIP':
print(f"{row['이름']}님, 특별 VIP 할인 쿠폰이 발급되었습니다.")
elif row['세그먼트'] == '우수고객':
print(f"{row['이름']}님, 우수고객 감사 이벤트에 초대합니다.")
else:
print(f"{row['이름']}님, 첫 구매 할인 쿠폰을 보내드립니다.")

자동 보고서 생성

조건문과 반복문을 활용해 데이터를 분석하고 자동으로 보고서를 생성하는 예제입니다.

판매 데이터를 분석하여 요약 보고서를 생성하는 코드를 작성해줘
import pandas as pd
from datetime import datetime
# 월별 판매 데이터
판매_데이터 = pd.DataFrame({
'날짜': pd.date_range(start='2023-01-01', periods=12, freq='M'),
'제품A': [120, 150, 130, 160, 190, 210, 200, 180, 150, 140, 160, 190],
'제품B': [90, 100, 85, 110, 120, 130, 140, 135, 125, 115, 130, 140]
})
# 년월 컬럼 추가
판매_데이터['년'] = 판매_데이터['날짜'].dt.year
판매_데이터['월'] = 판매_데이터['날짜'].dt.month
판매_데이터['분기'] = (판매_데이터['월'] - 1) // 3 + 1
# 보고서 생성
보고서 = [f"판매 분석 보고서 - 생성일: {datetime.now().strftime('%Y-%m-%d')}\n"]
# 분기별 데이터 분석
for 분기 in [1, 2, 3, 4]:
분기_데이터 = 판매_데이터[판매_데이터['분기'] == 분기]
제품A_합계 = 분기_데이터['제품A'].sum()
제품B_합계 = 분기_데이터['제품B'].sum()
총합계 = 제품A_합계 + 제품B_합계
보고서.append(f"\n{분기}분기 판매 실적:")
보고서.append(f"- 제품A: {제품A_합계}개")
보고서.append(f"- 제품B: {제품B_합계}개")
보고서.append(f"- 총합계: {총합계}개")
# 목표 달성 여부 확인 (목표: 제품별 250개)
if 제품A_합계 >= 250 and 제품B_합계 >= 250:
보고서.append("✅ 분기 목표 달성: 두 제품 모두 목표 달성")
elif 제품A_합계 >= 250:
보고서.append("⚠️ 분기 목표 부분 달성: 제품A만 목표 달성")
elif 제품B_합계 >= 250:
보고서.append("⚠️ 분기 목표 부분 달성: 제품B만 목표 달성")
else:
보고서.append("❌ 분기 목표 미달성: 두 제품 모두 목표 미달")
# 보고서 파일로 저장
with open('판매보고서.txt', 'w', encoding='utf-8') as f:
f.write('\n'.join(보고서))
print("보고서가 생성되었습니다: 판매보고서.txt")
자동화의 핵심

조건문과 반복문, 데이터 구조를 결합하면 다양한 업무 시나리오에 대응할 수 있는 유연한 자동화 솔루션을 구축할 수 있습니다. 정기 보고서 생성, 데이터 변환, 파일 처리 등 다양한 업무를 효율적으로 자동화할 수 있습니다.
