기본 try-except-else-finally 구조
ChatGPT 프롬프트: "파이썬에서 try-except-else-finally 구조의 간단한 예제 코드를 보여주세요." 
try:
    user_input = int(input("숫자를 입력하세요: "))
    result = 100 / user_input
except ValueError:
    print("유효한 숫자를 입력해주세요.")
except ZeroDivisionError:
    print("0으로 나눌 수 없습니다.")
else:
    print(f"계산 결과: {result}")
finally:
    print("작업이 완료되었습니다.")
※ 핵심 포인트: else 블록은 예외가 발생하지 않았을 때만 실행되며, finally 블록은 예외 발생 여부와 상관없이 항상 실행됩니다.

try:
    data = {"name": "홍길동"}
    print(data["age"])
    result = 10 / 0
except (KeyError, ZeroDivisionError, TypeError) as e:
    print(f"오류 발생: {type(e).__name__}")
    print(f"오류 메시지: {e}")
    
    # 예외 유형에 따른 세부 처리
    if isinstance(e, KeyError):
        print("존재하지 않는 키에 접근했습니다.")
    elif isinstance(e, ZeroDivisionError):
        print("0으로 나눌 수 없습니다.")
※ 핵심 포인트: 여러 예외 유형을 튜플로 묶어 한 번에 처리할 수 있으며, 예외 인스턴스를 변수에 저장해 활용할 수 있습니다.
사용자 정의 예외 클래스
ChatGPT 프롬프트: "파이썬에서 업무 자동화를 위한 사용자 정의 예외 클래스를 만드는 방법을 알려주세요." 
class MinimumBalanceError(Exception):
    """잔액이 최소 금액보다 적을 때 발생하는 예외"""
    def __init__(self, balance, minimum):
        self.balance = balance
        self.minimum = minimum
        self.message = f"잔액({balance:,}원)이 최소 금액({minimum:,}원)보다 적습니다."
        super().__init__(self.message)

def withdraw(balance, amount, minimum=1000):
    if balance - amount < minimum:
        raise MinimumBalanceError(balance - amount, minimum)
    return balance - amount

try:
    new_balance = withdraw(5000, 4500)
    print(f"출금 후 잔액: {new_balance}원")
except MinimumBalanceError as e:
    print(f"오류: {e}")
※ 핵심 포인트: Exception 클래스를 상속받아 목적에 맞는 예외 클래스를 직접 정의할 수 있습니다. 이는 업무 로직에 특화된 예외 처리에 유용합니다.

def process_excel_files(directory):
    # 오류 로그 파일 준비
    log_file = f"error_log_{datetime.now().strftime('%Y%m%d_%H%M%S')}.txt"

    for filename in os.listdir(directory):

            file_path = os.path.join(directory, filename)
            try:
                # 파일 읽기 시도
                df = pd.read_excel(file_path)
                
                # 여기서 데이터 처리 작업 수행
                # ...
                
                success_count += 1
                print(f"처리 완료: {filename}")
                
            except Exception as e:
                # 오류 발생 시 로그에 기록하고 다음 파일로
                error_count += 1
                with open(log_file, 'a') as f:
                    f.write(f"{filename}: {type(e).__name__} - {str(e)}\n")
