다중 프로그램 통합 자동화

여러 프로그램을 오가며 데이터를 주고받는 복잡한 워크플로우를 자동화합니다.

import pyautogui as pag
import time
import pyperclip
import os

def automate_data_transfer():
    # 엑셀에서 데이터 복사
    excel_app = pag.getWindowsWithTitle('Excel')[0]
    excel_app.activate()
    time.sleep(0.5)
    
    # 셀 선택 및 복사
    pag.click(x=200, y=300)
    pag.hotkey('ctrl', 'c')
    data = pyperclip.paste()
    
    # 웹 브라우저로 전환
    browser = pag.getWindowsWithTitle('Chrome')[0]
    browser.activate()
    time.sleep(0.5)
    
    # 데이터 검색
    pag.click(x=500, y=80)  # 주소창 클릭
    pag.hotkey('ctrl', 'a')  # 전체 선택
    pag.write('https://example.com/search')
    pag.press('enter')
    time.sleep(2)
    
    # 검색창에 데이터 입력
    search_btn = pag.locateCenterOnScreen('search_icon.png')
    if search_btn:
        pag.click(search_btn)
        pag.write(data)
        pag.press('enter')
        
    # 결과 스크린샷으로 저장
    time.sleep(3)
    pag.screenshot('search_result.png')

조건부 자동화 로직

화면 상태를 인식하고 상황에 따라 다른 작업을 수행하는 지능형 자동화를 구현합니다.

import pyautogui as pag
import time
import logging

# 로깅 설정
logging.basicConfig(level=logging.INFO, 
                   format='%(asctime)s - %(message)s',
                   filename='automation.log')

def smart_automation():
    # 상태 확인 및 조건부 처리
    try:
        # 로그인 상태 확인
        login_btn = pag.locateOnScreen('login_button.png', 
                                     confidence=0.8)
        if login_btn:
            logging.info("로그인 필요 상태 감지")
            pag.click(login_btn)
            time.sleep(1)
            
            # 아이디/비밀번호 입력
            pag.write('username')
            pag.press('tab')
            pag.write('password')
            pag.press('enter')
            time.sleep(3)
        
        # 업데이트 창 확인 및 처리
        update_popup = pag.locateOnScreen('update_popup.png', 
                                        confidence=0.8)
        if update_popup:
            logging.info("업데이트 알림창 감지")
            # 나중에 업데이트 버튼 클릭
            later_btn = pag.locateOnScreen('later_button.png')
            if later_btn:
                pag.click(later_btn)
            else:
                # 찾지 못하면 ESC로 닫기 시도
                pag.press('esc')
            time.sleep(1)
        
        # 메인 작업 수행
        main_btn = pag.locateOnScreen('main_button.png')
        if main_btn:
            logging.info("주요 기능 버튼 감지 및 클릭")
            pag.click(main_btn)
            # 추가 작업 계속...
        else:
            logging.error("주요 버튼을 찾을 수 없음")
            
    except Exception as e:
        logging.error(f"자동화 오류: {str(e)}")
        # 오류 스크린샷 저장
        pag.screenshot('error_state.png')
