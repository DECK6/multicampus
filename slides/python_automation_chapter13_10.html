# ChatGPT Prompt: "Show how to automate business card creation using Excel data and PowerPoint template"
import openpyxl from pptx import Presentation from pptx.util import Inches, Pt def create_business_cards(): # 엑셀에서 직원 정보 읽기 wb = openpyxl.load_workbook('employee_data.xlsx') ws = wb.active # PPT 템플릿 로드 prs = Presentation('business_card_template.pptx') for row in ws.iter_rows(min_row=2): # 새 슬라이드 추가 slide = prs.slides.add_slide(prs.slide_layouts[6]) # 직원 정보 추출 name = row[0].value position = row[1].value email = row[2].value phone = row[3].value # 텍스트 상자에 정보 입력 for shape in slide.shapes: if shape.name == 'name_box': shape.text_frame.text = name elif shape.name == 'position_box': shape.text_frame.text = position # 저장 prs.save('business_cards.pptx')
