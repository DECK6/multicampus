Outlook을 활용한 이메일 자동화

ChatGPT Prompt: "Create Python script for Outlook email automation" 
import win32com.client
import os

def send_outlook_email(to_email, subject, body, attachments=None):
    try:
        # Outlook 애플리케이션 객체 생성
        outlook = win32com.client.Dispatch('Outlook.Application')
        mail = outlook.CreateItem(0)  # 0 = Email item

        # 기본 정보 설정
        mail.To = to_email
        mail.Subject = subject
        mail.HTMLBody = body

        # 첨부파일 추가
        if attachments:
            for file in attachments:
                if os.path.exists(file):
                    mail.Attachments.Add(file)

        # 이메일 전송
        mail.Send()
        return True, "이메일이 성공적으로 전송되었습니다."

    except Exception as e:
        return False, f"이메일 전송 실패: {str(e)}"
