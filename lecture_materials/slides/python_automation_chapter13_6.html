# ChatGPT Prompt: "Show advanced PowerPoint automation with charts and images"
from pptx import Presentation from pptx.chart.data import ChartData from pptx.enum.chart import XL_CHART_TYPE from pptx.util import Inches prs = Presentation() slide = prs.slides.add_slide(prs.slide_layouts[5]) # 차트 데이터 설정 chart_data = ChartData() chart_data.categories = ['Q1', 'Q2', 'Q3', 'Q4'] chart_data.add_series('매출', (15.2, 18.7, 21.3, 25.9)) chart_data.add_series('이익', (8.1, 9.5, 11.2, 13.8)) # 차트 추가 x, y, cx, cy = Inches(2), Inches(2), Inches(6), Inches(4.5) chart = slide.shapes.add_chart( XL_CHART_TYPE.COLUMN_CLUSTERED, x, y, cx, cy, chart_data ).chart # 차트 제목 설정 chart.has_title = True chart.chart_title.text_frame.text = '2023년 분기별 실적'
