Twilio를 활용한 SMS 자동화

ChatGPT Prompt: "Create Python script for sending SMS using Twilio" 
from twilio.rest import Client

def send_sms(account_sid, auth_token, from_number, to_number, message):
    try:
        # Twilio 클라이언트 초기화
        client = Client(account_sid, auth_token)

        # SMS 전송
        message = client.messages.create(
            body=message,
            from_=from_number,
            to=to_number
        )
        
        return True, f"메시지가 성공적으로 전송됨 (SID: {message.sid})"
    
    except Exception as e:
        return False, f"메시지 전송 실패: {str(e)}"

# 사용 예시
credentials = {
    "account_sid": "your_account_sid",
    "auth_token": "your_auth_token",
    "from_number": "+1234567890",  # Twilio 번호
    "to_number": "+0987654321"     # 수신자 번호
}
