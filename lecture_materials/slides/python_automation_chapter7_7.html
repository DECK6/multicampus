행동 5: 데이터 넣기

웹 폼에 데이터를 입력하는 것은 웹 자동화의 핵심 기능 중 하나입니다. Selenium은 다양한 종류의 폼 요소에 데이터를 입력하고 조작하는 방법을 제공합니다.

텍스트 입력하기

텍스트 입력란에 데이터를 입력하는 가장 기본적인 방법은 send_keys() 메서드를 사용하는 것입니다.

# ChatGPT: "셀레니움으로 텍스트 입력하기"
from selenium import webdriver
from selenium.webdriver.common.by import By

# 요소 찾기
username_field = driver.find_element(By.ID, "username")
password_field = driver.find_element(By.NAME, "password")

# 기존 텍스트 지우기
username_field.clear()
password_field.clear()

# 텍스트 입력하기
username_field.send_keys("myusername")
password_field.send_keys("mypassword")

다양한 폼 요소 조작하기

체크박스 선택
checkbox = driver.find_element(By.ID, "terms")
if not checkbox.is_selected():
  checkbox.click()
라디오 버튼 선택
radio = driver.find_element(By.CSS_SELECTOR, "input[value='option2']")
radio.click()
드롭다운 선택
from selenium.webdriver.support.ui import Select

select = Select(driver.find_element(By.ID, "country"))
select.select_by_visible_text("대한민국")
파일 업로드
file_input = driver.find_element(By.ID, "file-upload")
file_input.send_keys("/path/to/file.jpg")

폼 제출하기

폼 데이터를 모두 입력한 후 제출하는 방법에는 여러 가지가 있습니다.

# ChatGPT: "셀레니움으로 폼 제출하기"
# 방법 1: 제출 버튼 찾아서 클릭하기
submit_button = driver.find_element(By.CSS_SELECTOR, "button[type='submit']")
submit_button.click()

# 방법 2: 폼 요소 찾아서 submit() 메서드 사용하기
form = driver.find_element(By.ID, "login-form")
form.submit()

# 방법 3: Enter 키 입력으로 제출하기
from selenium.webdriver.common.keys import Keys

password_field = driver.find_element(By.NAME, "password")
password_field.send_keys("mypassword" + Keys.ENTER)
특수 키 입력하기

텍스트 외에도 다양한 키보드 키를 입력할 수 있습니다.

from selenium.webdriver.common.keys import Keys

# Ctrl+A로 전체 선택 후 삭제
element.send_keys(Keys.CONTROL + "a")
element.send_keys(Keys.DELETE)

# TAB 키로 다음 필드로 이동
element.send_keys(Keys.TAB)

실제 폼 작성 예제

# ChatGPT: "셀레니움으로 회원가입 폼 작성하기"
def fill_registration_form(driver, user_data):
  # 페이지 접속
  driver.get("https://example.com/register")
  
  # 기본 정보 입력
  driver.find_element(By.ID, "first-name").send_keys(user_data["first_name"])
  driver.find_element(By.ID, "last-name").send_keys(user_data["last_name"])
  driver.find_element(By.ID, "email").send_keys(user_data["email"])
  
  # 비밀번호 입력
  driver.find_element(By.ID, "password").send_keys(user_data["password"])
  driver.find_element(By.ID, "confirm-password").send_keys(user_data["password"])
  
  # 드롭다운 선택
  select = Select(driver.find_element(By.ID, "country"))
  select.select_by_visible_text(user_data["country"])
  
  # 약관 동의 체크
  terms_checkbox = driver.find_element(By.ID, "terms")
  if not terms_checkbox.is_selected():
    terms_checkbox.click()
  
  # 폼 제출
  driver.find_element(By.CSS_SELECTOR, "button[type='submit']").click()
