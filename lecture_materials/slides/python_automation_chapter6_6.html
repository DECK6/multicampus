웹 요소와 상호작용하기
Selenium으로 버튼 클릭, 텍스트 입력 등 웹 요소와 상호작용하는 방법
Selenium으로 웹페이지의 요소와 상호작용하는 방법을 예시 코드와 함께 알려줘

클릭 및 버튼 제어

웹 요소를 찾아 클릭하거나 버튼을 눌러 웹페이지와 상호작용하는 방법입니다.

Python from selenium import webdriver
from selenium.webdriver.common.by import By

# 버튼 요소 찾기 및 클릭
login_button = driver.find_element(By.ID, "login")
login_button.click()

# CSS 선택자로 요소 찾아 클릭
menu_item = driver.find_element(By.CSS_SELECTOR, ".menu-item")
menu_item.click()

# 링크 텍스트로 요소 찾아 클릭
signup_link = driver.find_element(By.LINK_TEXT, "회원가입")
signup_link.click()

텍스트 입력

검색창, 로그인 폼 등에 텍스트를 입력하는 방법입니다.

Python # 입력 필드 찾기 및 텍스트 입력
username = driver.find_element(By.NAME, "username")
username.send_keys("user123")

# 비밀번호 입력
password = driver.find_element(By.ID, "password")
password.send_keys("securepassword")

# 입력 필드 내용 지우기
search_box = driver.find_element(By.ID, "search")
search_box.clear()
search_box.send_keys("파이썬 자동화")

폼 요소 제어

드롭다운 메뉴, 체크박스, 라디오 버튼 등의 폼 요소를 제어하는 방법입니다.

Python from selenium.webdriver.support.ui import Select

# 드롭다운 메뉴 조작
dropdown = Select(driver.find_element(By.ID, "country"))

# 옵션 선택 방법
dropdown.select_by_visible_text("대한민국") # 보이는 텍스트로 선택
dropdown.select_by_value("KR") # 값으로 선택
dropdown.select_by_index(3) # 인덱스로 선택

# 체크박스 선택/해제
checkbox = driver.find_element(By.ID, "agree")
if not checkbox.is_selected():
    checkbox.click()
