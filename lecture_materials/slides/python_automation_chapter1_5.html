Pandas를 활용한 데이터 처리
Pandas는 데이터 분석과 처리를 위한 파이썬 라이브러리로, 고성능의 데이터 구조와 데이터 분석 도구를 제공합니다. 대용량 데이터 처리, 데이터 정제, 통계 분석 등 다양한 데이터 작업을 직관적이고 효율적으로 수행할 수 있습니다.
Series
1차원 배열 형태의 데이터 구조로, 인덱스와 값으로 구성됩니다. 라벨이 있는 1차원 배열로 생각할 수 있습니다.
import pandas as pd s = pd.Series([10, 20, 30, 40], index=['a', 'b', 'c', 'd']) print(s)
DataFrame
2차원 테이블 형태의 데이터 구조로, 여러 Series가 모인 형태입니다. 행과 열이 있는 스프레드시트와 유사합니다.
data = { '이름': ['김철수', '이영희', '박민수'], '나이': [25, 28, 31], '직업': ['개발자', '디자이너', '마케터'] } df = pd.DataFrame(data) print(df)

# Pandas를 활용한 데이터 분석 예제 import pandas as pd import matplotlib.pyplot as plt # CSV 파일에서 판매 데이터 읽기 sales_data = pd.read_csv('sales_data.csv') # 데이터 확인 print("데이터 미리보기:") print(sales_data.head()) # 데이터 정보 확인 print("\n데이터 정보:") print(sales_data.info()) # 날짜 열을 datetime 형식으로 변환 sales_data['날짜'] = pd.to_datetime(sales_data['날짜']) # 월별 매출 계산 sales_data['월'] = sales_data['날짜'].dt.month monthly_sales = sales_data.groupby('월')['매출액'].sum().reset_index() # 제품별 판매량 계산 product_sales = sales_data.groupby('제품명')['판매량'].sum().sort_values(ascending=False) # 결과 출력 print("\n월별 매출:") print(monthly_sales) print("\n제품별 판매량:") print(product_sales)

# 데이터 정제 예제 import pandas as pd # 데이터 로드 df = pd.read_excel('고객_데이터.xlsx') # 결측치 확인 print("결측치 개수:") print(df.isnull().sum()) # 결측치 처리 # 1. 수치형 데이터는 평균값으로 대체 df['나이'] = df['나이'].fillna(df['나이'].mean()) # 2. 범주형 데이터는 최빈값으로 대체 df['직업'] = df['직업'].fillna(df['직업'].mode()[0]) # 3. 특정 열은 결측치가 있는 행 제거 df = df.dropna(subset=['고객ID', '이메일']) # 중복 데이터 확인 duplicate_count = df.duplicated().sum() print(f"\n중복 데이터 수: {duplicate_count}") # 중복 데이터 제거 df = df.drop_duplicates() print(f"\n처리 후 데이터 크기: {df.shape}")
