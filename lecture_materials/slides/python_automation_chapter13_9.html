# ChatGPT Prompt: "Show how to create and edit images with Pillow for business cards"
from PIL import Image, ImageDraw, ImageFont # 명함 이미지 생성 def create_business_card(name, title, company, email, phone): # 새 이미지 생성 (가로 900px, 세로 500px) card = Image.new('RGB', (900, 500), color='white') draw = ImageDraw.Draw(card) # 폰트 로드 name_font = ImageFont.truetype('arial.ttf', 60) info_font = ImageFont.truetype('arial.ttf', 40) # 텍스트 추가 draw.text((50, 100), name, fill='black', font=name_font) draw.text((50, 200), title, fill='gray', font=info_font) draw.text((50, 280), company, fill='black', font=info_font) draw.text((50, 340), email, fill='blue', font=info_font) draw.text((50, 400), phone, fill='black', font=info_font) return card
