여러 Excel 파일 병합 자동화 시간 절약

여러 개의 Excel 파일에서 데이터를 추출하여 하나의 파일로 병합하는 자동화 스크립트입니다.

여러 지점의 월간 판매 보고서 엑셀 파일을 하나로 통합하는 코드를 작성해줘
import pandas as pd
import glob
import os
# 모든 지점 보고서 파일 가져오기
보고서_폴더 = './월간보고서/'
보고서_파일목록 = glob.glob(보고서_폴더 + '*.xlsx')
# 모든 데이터를 저장할 빈 데이터프레임
통합_데이터 = pd.DataFrame()
for 파일경로 in 보고서_파일목록:
지점명 = os.path.basename(파일경로).split('_')[0]
print(f"{지점명} 보고서 처리 중...")
# 엑셀 파일 읽기
df = pd.read_excel(파일경로, sheet_name='판매데이터')
# 지점 정보 추가
df['지점'] = 지점명
# 통합 데이터에 추가
통합_데이터 = pd.concat([통합_데이터, df], ignore_index=True)
# 피벗 테이블 생성 (지점별, 상품별 판매량 집계)
피벗_데이터 = pd.pivot_table(
통합_데이터,
values='판매량',
index='상품명',
columns='지점',
aggfunc='sum',
fill_value=0
)
# 합계 행 추가
피벗_데이터.loc['총합계'] = 피벗_데이터.sum()
# 결과 저장
with pd.ExcelWriter('통합월간보고서.xlsx') as writer:
통합_데이터.to_excel(writer, sheet_name='전체데이터', index=False)
피벗_데이터.to_excel(writer, sheet_name='지점별요약')
print("통합 보고서가 생성되었습니다.")
서식 있는 보고서 생성

openpyxl을 사용하여 서식이 적용된 전문적인 Excel 보고서를 자동으로 생성할 수 있습니다.

openpyxl로 서식이 있는 판매 보고서를 자동 생성하는 코드를 작성해줘
import openpyxl
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.chart import BarChart, Reference
import pandas as pd
# 샘플 데이터 생성
데이터 = {
'상품': ['노트북', '태블릿', '스마트폰', '모니터', '키보드'],
'판매량': [52, 28, 75, 35, 48],
'단가': [1200000, 800000, 950000, 350000, 90000]
}
df = pd.DataFrame(데이터)
df['매출액'] = df['판매량'] * df['단가']
# 새 워크북 생성
wb = openpyxl.Workbook()
ws = wb.active
ws.title = "제품별 판매 현황"
# 제목 추가
ws['A1'] = "2023년 3분기 제품별 판매 현황"
ws.merge_cells('A1:E1')
ws['A1'].font = Font(size=16, bold=True, color="4B8BBE")
ws['A1'].alignment = Alignment(horizontal='center')
# 헤더 추가
헤더 = ['제품명', '판매수량', '단가', '매출액', '비고']
for col_num, 헤더명 in enumerate(헤더, 1):
cell = ws.cell(row=3, column=col_num, value=헤더명)
cell.font = Font(bold=True, color="FFFFFF")
cell.fill = PatternFill("solid", fgColor="4B8BBE")
cell.alignment = Alignment(horizontal='center')
# 데이터 추가 및 서식 지정
for i, row in enumerate(df.itertuples(), 1):
row_num = i + 3
ws.cell(row=row_num, column=1).value = row.상품
ws.cell(row=row_num, column=2).value = row.판매량
ws.cell(row=row_num, column=3).value = row.단가
ws.cell(row=row_num, column=4).value = row.매출액
# 조건부 서식 (매출액이 높은 경우 표시)
if row.매출액 > 50000000:
ws.cell(row=row_num, column=5).value = "우수"
ws.cell(row=row_num, column=5).font = Font(color="FF0000", bold=True)
# 차트 생성
chart = BarChart()
데이터_영역 = Reference(ws, min_row=3, max_row=8, min_col=2, max_col=2)
카테고리 = Reference(ws, min_row=4, max_row=8, min_col=1)
chart.add_data(데이터_영역, titles_from_data=True)
chart.set_categories(카테고리)
chart.title = "제품별 판매수량"
ws.add_chart(chart, "G4")
# 파일 저장
wb.save('판매보고서.xlsx')
