# ChatGPT Prompt: "Show basic Word document automation with python-docx"
from docx import Document from docx.shared import Pt, RGBColor from docx.enum.text import WD_ALIGN_PARAGRAPH # 새 문서 생성 doc = Document() # 제목 추가 title = doc.add_heading('업무 보고서', level=0) title.alignment = WD_ALIGN_PARAGRAPH.CENTER # 문단 추가 doc.add_paragraph('작성일: 2024년 1월 15일') doc.add_paragraph('작성자: 홍길동') # 목차 추가 doc.add_heading('목차', level=1) points = ['주요 성과', '향후 계획', '건의사항'] for point in points: p = doc.add_paragraph() p.add_run('• ' + point) # 문서 저장 doc.save('업무보고서.docx')
